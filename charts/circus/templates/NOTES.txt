To access the Usher service:
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/component=usher,app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  echo "Visit http://127.0.0.1:3000 to access Usher"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 3000:3000

To access MinIO Console:
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/component=minio,app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  echo "Visit http://127.0.0.1:9001 to access MinIO Console"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 9001:9001

Note: Make sure to set the Anthropic API key secret before deploying:
  kubectl create secret generic anthropic-api-key --from-literal=api-key=YOUR_API_KEY
